## 8.获取帮助信息

## 10.[a-k]*  //以a-k之间任意字符开头的

## 12.输出重定向

    >>  //追加到原有内容之后
## 13.查看文件内容
    cat  //查看文件内容
    more  //根据终端大小适应性查看文件内容

## 14.管道
    |  //ls -lha | more
## 15.clear  //清屏
## 16.gedit  //与vi、vim类似的编辑工具
## 17.文件夹操作
    mkdir  //创建文件夹
    mkdir 文件路径 -p  //递归创建文件夹
    rmdir  //删除非空文件夹
## 18.文件操作   
    rm [-r]  //递归删除目录下文件（用于删除非空文件夹）
    rm [-f]  //强制删除，忽略是否存在
    rm [-i]  //以交互方式执行
## 19.硬链接和软连接
    ln -s 源文件 链接文件  //软连接
    ln 源文件 链接文件    //硬链接
    软连接删除源文件以后，链接文件失效，其中链接文件类似快捷方式
    硬链接相当于同一个文件不同名字，删除源文件只是删除了其中一个名字，只有当所有名字都删完才会清空文件
## 20.合并文件
    cat 1.txt 2.txt > n.txt
## 21.查找文件中的字符串
    grep [-n] '文件名'  //-n显示行数，文件名可使用正则表达式
    如：'[Hh]ello'  //hello或Hello
    '^abc'  //以abc开头的
    'abc$'  //以abc结尾的
    //cat 文件 | grep 字符串  可以实现对文件内容快速查找定位
## 22.搜索文件
    find 路径 -name 文件名   //可使用正则表达式
              -size 文件大小 //-size +4k -size -5M 大于4k小于5M的文件
              -perm 权限     //0777
    //路径表示从哪里开始搜索
## 23.tree  //文件结构以树的形式呈现
## 24.复制拷贝
    cp -a  //保持原文件属性
       -f  //已经存在目标文件，不提示
       -r  //递归复制（用于文件夹）
       -v  //显示拷贝进度
       -i  //交互式复制
## 25.移动剪切
    mv -f  //禁止交互，覆盖也不提示
       -i  //交互式
       -v  //显示进度
    //可用于对文件重命名
## 26.打包、压缩与解压缩
### 打包
    tar -c  //创建打包文件（creat）
        -v  //显示进度
        -f  //指定档案名，该参数必须放在最后，后跟tar包
        -t  //列出档案中包含的文件
        -x  //解包文件
    //tar -cvf test.tar [文件1] [文件2]...
### 打包并压缩
#### 使用gzip压缩
    gzip -d 源文件 [目标压缩文件]  //解压文件
         -r                      //压缩所有子目录
    tar -zcvf 文件名.tar.gz 被压缩文件1 被压缩文件2...//被压缩文件用空格分隔
        -zxvf 压缩文件 [-C] [指定路径]  //解压文件
#### 使用bzip压缩
    tar -jcvf 文件名.tar.bz 被压缩文件1 被压缩文件2...//压缩
        -jxvf  //解压文件
#### 使用bzip压缩
    zip [-r] 目标文件 源文件  //目标文件不用添加扩展名
    unzip -d 解压后目录文件 压缩文件
## 27.which 命令  //查看命令的绝对路径

# Linux系统管理
## 快捷操作
    Ctrl + Shift + T  //新建一个标签
    Ctrl + Alt + T    //快速打开一个terminal
## 2.日历和时间
    cal  //查看当前日历
    date  //查看当前时间
    date [MMDDhhmm[[CC]YY][.SS]] + format  //修改时间
    其中:MM月、DD日、hh小时、mm分钟、CC年份前两位、YY年份后四位、SS秒
    date '+%y,%m,%d,%H,%M,%S'  //显示时间格式
    其中%y或%Y年，后面依次是月、日、时、分、秒
## 3.查看进程信息
    ps -aux
       -a  //显示终端上所有进程，包含其他用户进程
       -u  //显示进程详细状态
       -x  //显示没有控制终端的进程
       -w  //显示加宽以便显示更多信息
       -r  //只显示正在运行的进程
    top [-d] //动态显示进程，-d指定更新时间间隔
    在top命令执行以后，按下以下按键可对显示结果进行操作：
    M  //根据内存使用量排序
    P  //根据cpu占有率排序
    T  //根据进程运行时间长短排序
    U  //根据后面输入的用户名筛选
    K  //根据后面输入的PID杀进程
    q  //退出
    h  //获得帮助
## 4.终止进程
    kill [-signal] pid  //signal=9 强制终止，配合ps使用查看进程id
## 5.关机重启
    reboot  //重新启动系统
    shutdown -r now  //重启系统，并给别的用户提示
             -h now  //立刻关机，其中now相当于时间为0
             -h 20:25  //系统在今天20:25关机
             -h +10  //系统十分钟后自动关机
    init [0-6]  //0关机、1单用户模式、2-5多用户、6重启
## 6.检测磁盘空间
    df -a  //显示所有文件系统磁盘使用情况
       -m  //以MB为单位显示
       -t  //显示各指定文件系统的磁盘使用情况
       -T  //显示文件系统
       -lh //以列表形式带单位显示
## 7.检测目录所占磁盘空间
    du [选项] 目录或文件
       -a  //递归显示指定目录中各文件和子目录中文件占用大小
       -s  //显示指定文件或目录占用大小
       -b  //以字节为单位显示磁盘使用情况
       -l  //计算所有文件大小，对硬链接文件会计算多次
## 8.快捷操作
    Ctrl + Alt + F1~F6  //命令行模式  /etc/inittab
                 F7~F12 //图形界面    /etc/x11/xinit/xinitrc
## 9.who指令
    who [-q]  //谁登录过系统，-q 有几个用户
    whoami    //当前登录用户是谁
## 10.sudo指令
    sudo -s  //切换到 root
## 11.passwd 用户名  //修改用户密码（root权限）
## 12.添加用户账号
    useradd 用户名 -m  //自动创建用户目录
                  -d 目录路径
                  -g 指定组名称
注:1.useradd 用户名 -m = adduser 用户名
   2.建议使用 adduser 按引导设置
   3.cat /etc/passwd 可查看系统中用户情况
   4.普通用户不在 sudoer 中，不能使用 sudo 指令
## 13.删除用户
    userdel 用户名 [-r]  //-r 同时删除用户主目录
## 14.切换用户
    su [-] 用户名  //加 - 切换用户后将进入该用户主目录
## 15.查看用户分组
    groups 用户名
    在adm和sudo组的用户就可以使用sudo指令
## 16.查看有哪些用户组
    cat /etc/group
    groupmod + 三次tab键
## 17.添加、删除组
    groupadd/groupdel
## 18.修改用户所在组
    usermod -g 用户组 用户名  //修改用户默认分组
            -a -G            //为用户添加其他组
## 19.为创建的普通用户添加sudo权限
    sudo usermod -a -G adm 用户名
    sudo usermod -a -G sudo 用户名
## 20.修改文件权限
     _ | _ _ _ | _ _ _ | _ _ _  // r w x对应二进制换算为4 2 1
文件类型|所有者(u)|同组(g)|其他用户(o) 
### 字母法
chmod [u,g,o][=,+,-][r,w,x] 文件名  //中间用逗号分隔
### 数字法
chmod [u,g,o]=[0-7] 文件名
chmod 777 文件名  //三位数从左到右依次对应u,g,o
## 21.修改文件所有者
    chown 用户名 文件或目录
## 22.修改文件所属组
    chgrp 用户组名 文件或目录
## 23.更新源
    gedit /etc/apt/sources.list
    apt-get update

# Ubuntu软件操作指令
## 1.apt-get 
    apt-get install package [--reinstall]  //安装包[重新安装包]    
            -f install  //修复安装
            remove package  //删除包
            remove package --purge  //删除包和配置文件等
            build-dep package  //安装相关编译环境
            upgrade  //更新已安装的包
            dist-upgrade  //升级系统
            source package  //下载该包源代码
            clean && apt-get autoclean  //清理无用的包
            check  //检查是否有损坏的依赖
## 2.apt-cache
    apt-cache search package  //搜索包
              show package  //获取包的相关信息，大小、版本等
              depends package  //了解该包依赖哪些包
              rdepends package  //查看该包被哪些包依赖
